## WHAT THIS PROJECT IS

CricketMind is a domain-adapted language model built on NVIDIA's Nemotron-Mini-4B-Instruct,
fine-tuned for two tasks:
1. Cricket rules reasoning (Laws of Cricket)
2. Match situation analysis

**Techniques used (all three required by NVIDIA JD):**
- Response Distillation — Claude as teacher model generating reasoning traces
- QLoRA (r=16) — Low-rank adaptation for efficient fine-tuning
- 4-bit NF4 Quantization — Memory-efficient training on A100

**Target:** NVIDIA Open Models PM role application portfolio

## TASK 6: CREATE scripts/03_generate_laws_qa.py

This script uses the Anthropic API to generate QA pairs from the laws.
Claude Code writes the complete script.

```python
#!/usr/bin/env python3
"""
03_generate_laws_qa.py
Uses Claude as generator to create QA training pairs from MCC Laws.
Run: python scripts/03_generate_laws_qa.py
Requires: ANTHROPIC_API_KEY in environment
Output: data/raw_laws_qa.json
"""

import json
import time
import os
import sys
from pathlib import Path
from dotenv import load_dotenv
import anthropic

load_dotenv()

ANTHROPIC_API_KEY = os.getenv("ANTHROPIC_API_KEY")
if not ANTHROPIC_API_KEY:
    print("ERROR: ANTHROPIC_API_KEY not set. Add it to .env file.")
    sys.exit(1)

client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)

DATA_DIR = Path("data")
LAWS_FILE = DATA_DIR / "laws" / "mcc_laws_text.json"
OUTPUT_FILE = DATA_DIR / "raw_laws_qa.json"

QA_GENERATION_PROMPT = """You are building a high-quality training dataset for a cricket rules reasoning model called CricketMind.

Given the following cricket law summary:
<law>
Law {law_number}: {law_title}
{law_clauses}
</law>

Generate exactly 20 training examples as a JSON array. Each example must follow this exact format:
{{
  "instruction": "You are CricketMind, an expert in the Laws of Cricket. Answer the following question accurately and reason step by step.",
  "input": "<specific question about this law>",
  "output": "<detailed answer with numbered reasoning chain>"
}}

Generate exactly 5 examples of each type:
TYPE 1 — DIRECT RECALL: "Under Law {law_number}, what is [specific rule or definition]?"
TYPE 2 — SCENARIO REASONING: Multi-condition scenario requiring 2-3 step logic chain
TYPE 3 — MISCONCEPTION: "A player believes that [common wrong assumption]. Is this correct?"
TYPE 4 — EDGE CASE: "What happens if [unusual combination of circumstances]?"

STRICT RULES:
- Every output MUST include a numbered reasoning chain (1. 2. 3.), not just the final answer
- Every claim must be accurate to the actual Laws of Cricket
- Do NOT hallucinate law numbers or sub-clauses not in the provided text
- Output ONLY valid JSON array — no preamble, no markdown, no explanation
- Minimum output length per example: 150 characters"""


def generate_qa_for_law(law: dict) -> list:
    clauses_text = "\n".join(f"  - {c}" for c in law["key_clauses"])
    
    prompt = QA_GENERATION_PROMPT.format(
        law_number=law["number"],
        law_title=law["title"],
        law_clauses=clauses_text
    )
    
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1000,
        messages=[{"role": "user", "content": prompt}]
    )
    
    text = response.content[0].text.strip()
    
    # Strip markdown fences if present
    if text.startswith("```"):
        text = text.split("```")[1]
        if text.startswith("json"):
            text = text[4:]
    text = text.strip()
    
    examples = json.loads(text)
    return examples


def main():
    print("=" * 60)
    print("CricketMind — Laws QA Generation")
    print("=" * 60)
    
    with open(LAWS_FILE) as f:
        laws_data = json.load(f)
    
    laws = laws_data["laws"]
    print(f"Loaded {len(laws)} laws. Generating 20 QA pairs each...\n")
    
    all_examples = []
    failed_laws = []
    
    for i, law in enumerate(laws):
        print(f"[{i+1}/{len(laws)}] Law {law['number']}: {law['title']}")
        
        for attempt in range(3):
            try:
                examples = generate_qa_for_law(law)
                
                # Quality filter
                accepted = []
                for ex in examples:
                    if (
                        isinstance(ex, dict)
                        and "instruction" in ex
                        and "input" in ex
                        and "output" in ex
                        and len(ex["output"]) >= 150
                        and any(f"{n}." in ex["output"] for n in [1, 2, 3])
                    ):
                        ex["law_number"] = law["number"]
                        ex["type"] = "laws_qa"
                        accepted.append(ex)
                
                all_examples.extend(accepted)
                print(f"  ✓ {len(accepted)}/{len(examples)} accepted")
                break
                
            except Exception as e:
                print(f"  Attempt {attempt+1} failed: {e}")
                if attempt == 2:
                    failed_laws.append(law["number"])
                    print(f"  ✗ Law {law['number']} skipped after 3 failures")
                time.sleep(2)
        
        time.sleep(0.8)  # rate limit buffer
    
    # Save
    DATA_DIR.mkdir(exist_ok=True)
    with open(OUTPUT_FILE, "w") as f:
        json.dump(all_examples, f, indent=2)
    
    print(f"\n{'='*60}")
    print(f"COMPLETE: {len(all_examples)} QA examples saved to {OUTPUT_FILE}")
    if failed_laws:
        print(f"WARNING: Failed laws: {failed_laws}")
    print(f"{'='*60}")


if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
02_distill_match_situations.py
Response distillation: Claude as teacher model generating structured
chain-of-thought reasoning traces for match situation analysis.
Run: python scripts/02_distill_match_situations.py
Requires: ANTHROPIC_API_KEY in environment
Output: data/distilled_match_situations.json
"""

import json
import time
import os
import sys
from pathlib import Path
from dotenv import load_dotenv
import anthropic

load_dotenv()

ANTHROPIC_API_KEY = os.getenv("ANTHROPIC_API_KEY")
if not ANTHROPIC_API_KEY:
    print("ERROR: ANTHROPIC_API_KEY not set. Add it to .env file.")
    sys.exit(1)

client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)

DATA_DIR = Path("data")
INPUTS_FILE = DATA_DIR / "match_situation_inputs.json"
OUTPUT_FILE = DATA_DIR / "distilled_match_situations.json"
MIN_OUTPUT_LENGTH = 200

TEACHER_SYSTEM = """You are a world-class cricket analyst and certified MCC rules official 
with 20 years of experience. You have comprehensive knowledge of all 42 Laws of Cricket 
and elite-level match strategy.

For the cricket scenario provided, respond in this EXACT structured format:

1. SITUATION ASSESSMENT: Identify all key variables at play
2. APPLICABLE LAWS: Cite specific Law numbers and sub-clauses if relevant (write "N/A - Strategic scenario" if purely strategic)
3. REASONING CHAIN: Step-by-step conditional logic leading to your conclusion
4. DECISION / RECOMMENDATION: Clear, unambiguous answer
5. CONFIDENCE LEVEL: High / Medium / Low — one sentence explaining why

Rules:
- Be technically precise. Do NOT hallucinate Law numbers.
- If multiple valid interpretations exist, address each and state which is correct.
- Prioritize accuracy over length. Keep under 500 words."""


def distill_scenario(scenario: str, attempt: int = 0) -> str | None:
    try:
        response = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1000,
            system=TEACHER_SYSTEM,
            messages=[{"role": "user", "content": scenario}]
        )
        return response.content[0].text.strip()
    except Exception as e:
        if attempt < 2:
            print(f"    Retry {attempt+1}: {e}")
            time.sleep(2)
            return distill_scenario(scenario, attempt + 1)
        print(f"    FAILED after 3 attempts: {e}")
        return None


def main():
    print("=" * 60)
    print("CricketMind — Response Distillation Pipeline")
    print("Teacher: Claude | Student target: Nemotron Mini 4B")
    print("=" * 60)
    
    with open(INPUTS_FILE) as f:
        scenarios = json.load(f)
    
    print(f"Loaded {len(scenarios)} scenarios. Running distillation...\n")
    
    accepted = []
    rejected = []
    
    for i, scenario in enumerate(scenarios):
        print(f"[{i+1}/{len(scenarios)}] {scenario[:70]}...")
        
        output = distill_scenario(scenario)
        
        if output is None:
            print(f"  ✗ REJECTED — API error")
            rejected.append({"scenario": scenario, "reason": "api_error"})
            continue
        
        if len(output) < MIN_OUTPUT_LENGTH:
            print(f"  ✗ REJECTED — too short ({len(output)} < {MIN_OUTPUT_LENGTH} chars)")
            rejected.append({"scenario": scenario, "reason": "too_short", "length": len(output)})
            continue
        
        # Check structure quality — must have at least 3 of 5 sections
        sections = ["SITUATION ASSESSMENT", "APPLICABLE LAWS", "REASONING CHAIN",
                    "DECISION", "CONFIDENCE"]

#!/usr/bin/env python3
"""

import json

from pathlib import Path

def clean_example(ex: dict) -> dict:
    """Strip metadata fields not needed for training."""
    return {
        "instruction": ex["instruction"],
        "input": ex["input"],
        "output": ex["output"]
    }


def main():
    print("=" * 60)
    print("CricketMind — Dataset Preparation")
    print("=" * 60)
    
    # Load layers
    layers = {}
    
    if LAWS_QA_FILE.exists():
        with open(LAWS_QA_FILE) as f:
            laws_data = json.load(f)
        layers["Laws QA"] = laws_data
        print(f"  Layer 1 — Laws QA:              {len(laws_data)} examples")
    else:
        print(f"  WARNING: {LAWS_QA_FILE} not found. Run 03_generate_laws_qa.py first.")

    if DISTILLED_FILE.exists():
        with open(DISTILLED_FILE) as f:

        layers["Distilled Match Situations"] = distilled_data

#!/usr/bin/env python3
"""
04_train_qlora.py
QLoRA fine-tuning of Nemotron-Mini-4B-Instruct on CricketMind dataset.
Run ON RUNPOD A100 — not locally.
Run: python scripts/04_train_qlora.py
Requires: data/train.json, data/val.json in working directory
Output: ./cricketmind-lora/
"""

import json
import torch
from pathlib import Path
from datasets import Dataset
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    BitsAndBytesConfig,
    TrainingArguments,
)
from peft import LoraConfig, get_peft_model, prepare_model_for_kbit_training
from trl import SFTTrainer

# ── Config ──────────────────────────────────────────────────────────────────
MODEL_NAME = "nvidia/Nemotron-Mini-4B-Instruct"
FALLBACK_MODEL = "meta-llama/Llama-3.2-3B-Instruct"  # use if Nemotron access fails
OUTPUT_DIR = "./cricketmind-lora"
TRAIN_FILE = "data/train.json"
VAL_FILE = "data/val.json"

# QLoRA config
LORA_R = 16
LORA_ALPHA = 32
LORA_DROPOUT = 0.05
TARGET_MODULES = ["q_proj", "v_proj"]

# Training config
NUM_EPOCHS = 3
BATCH_SIZE = 4
GRAD_ACCUMULATION = 4  # effective batch = 16
LEARNING_RATE = 2e-4
MAX_SEQ_LENGTH = 1024
WARMUP_RATIO = 0.03
# ────────────────────────────────────────────────────────────────────────────


def format_instruction(example: dict) -> str:
    return (
        f"### Instruction:\n{example['instruction']}\n\n"
        f"### Input:\n{example['input']}\n\n"
        f"### Response:\n{example['output']}"
    )


def main():
    print("=" * 60)
    print("CricketMind — QLoRA Fine-Tuning")
    print(f"GPU: {torch.cuda.get_device_name(0)}")
    print(f"VRAM: {torch.cuda.get_device_properties(0).total_memory / 1e9:.1f} GB")
    print("=" * 60)

    # Load data
    with open(TRAIN_FILE) as f:
        train_raw = json.load(f)
    with open(VAL_FILE) as f:
        val_raw = json.load(f)

    train_dataset = Dataset.from_list(train_raw)
    val_dataset = Dataset.from_list(val_raw)
    print(f"Train: {len(train_dataset)} | Val: {len(val_dataset)}")

    # 4-bit quantization
    bnb_config = BitsAndBytesConfig(
        load_in_4bit=True,
        bnb_4bit_use_double_quant=True,
        bnb_4bit_quant_type="nf4",
        bnb_4bit_compute_dtype=torch.bfloat16,
    )

    # Load model
    print(f"\nLoading {MODEL_NAME}...")
    try:
        tokenizer = AutoTokenizer.from_pretrained(MODEL_NAME, trust_remote_code=True)
        model = AutoModelForCausalLM.from_pretrained(
            MODEL_NAME,
            quantization_config=bnb_config,
            device_map="auto",
            trust_remote_code=True,
        )
    except Exception as e:
        print(f"Could not load {MODEL_NAME}: {e}")
        print(f"Falling back to {FALLBACK_MODEL}")
        tokenizer = AutoTokenizer.from_pretrained(FALLBACK_MODEL)
        model = AutoModelForCausalLM.from_pretrained(
            FALLBACK_MODEL,
            quantization_config=bnb_config,
            device_map="auto",
        )

    tokenizer.pad_token = tokenizer.eos_token
    tokenizer.padding_side = "right"
    model.config.use_cache = False

    # Apply LoRA
    model = prepare_model_for_kbit_training(model)
    lora_config = LoraConfig(
        r=LORA_R,
        lora_alpha=LORA_ALPHA,
        target_modules=TARGET_MODULES,
        lora_dropout=LORA_DROPOUT,
        bias="none",
        task_type="CAUSAL_LM",
    )
    model = get_peft_model(model, lora_config)
    model.print_trainable_parameters()

    # Training
    training_args = TrainingArguments(
        output_dir=OUTPUT_DIR,
        num_train_epochs=NUM_EPOCHS,
        per_device_train_batch_size=BATCH_SIZE,
        gradient_accumulation_steps=GRAD_ACCUMULATION,
        warmup_ratio=WARMUP_RATIO,
        learning_rate=LEARNING_RATE,
        bf16=True,
        logging_steps=10,
        evaluation_strategy="steps",
        eval_steps=50,
        save_strategy="steps",
        save_steps=50,
        load_best_model_at_end=True,
        report_to="none",
    )

    trainer = SFTTrainer(
        model=model,
        args=training_args,
        train_dataset=train_dataset,
        eval_dataset=val_dataset,
        formatting_func=format_instruction,
        max_seq_length=MAX_SEQ_LENGTH,
    )

    print("\nStarting training...")
    trainer.train()

    trainer.save_model(OUTPUT_DIR)
    tokenizer.save_pretrained(OUTPUT_DIR)

    print(f"\n{'='*60}")
    print(f"TRAINING COMPLETE. LoRA adapters saved to {OUTPUT_DIR}")
    print(f"Next step: Run 05_merge_and_export.py")
    print(f"{'='*60}")


if __name__ == "__main__":
    main()

## TASK 12: CREATE scripts/06_evaluate_cricketbench.py

```python
#!/usr/bin/env python3
"""
06_evaluate_cricketbench.py
Runs CricketBench evaluation against the trained model.
Run locally with Ollama after model is exported.
Run: python scripts/06_evaluate_cricketbench.py
Requires: ollama running locally with 'cricketmind' model created
Output: evaluation/results_v01.json, prints score summary
"""

import json
import subprocess
import sys
from pathlib import Path
from datetime import datetime

BENCH_FILE = Path("evaluation/cricketbench_v01.json")
RESULTS_FILE = Path("evaluation/results_v01.json")
MODEL_NAME = "cricketmind"

CATEGORY_WEIGHTS = {
    "laws_recall": 0.30,
    "conditional_reasoning": 0.35,
    "match_situation": 0.25,
    "edge_case": 0.10,
}

SCORING_GUIDE = """
Score each response:
  2 = Fully correct — right answer AND correct reasoning chain
  1 = Partially correct — right conclusion but incomplete/flawed reasoning
  0 = Incorrect — wrong answer or critically wrong reasoning

Key concepts to check are listed for each question.
"""


def ask_model(question: str) -> str:
    result = subprocess.run(
        ["ollama", "run", MODEL_NAME, question],
        capture_output=True,
        text=True,
        timeout=60
    )
    return result.stdout.strip()


def main():
    print("=" * 60)
    print("CricketMind — CricketBench v0.1 Evaluation")
    print("=" * 60)
    print(SCORING_GUIDE)

    with open(BENCH_FILE) as f:
        questions = json.load(f)

    # Check Ollama is running
    try:
        subprocess.run(["ollama", "list"], capture_output=True, check=True)
    except Exception:
        print("ERROR: Ollama is not running or not installed.")
        print("Install: https://ollama.com | Then: ollama create cricketmind -f model/Modelfile")
        sys.exit(1)

    results = []
    scores_by_category = {cat: [] for cat in CATEGORY_WEIGHTS}

    for i, q in enumerate(questions):
        print(f"\n{'─'*60}")
        print(f"[{i+1}/{len(questions)}] {q['id']} | {q['category']} | {q['difficulty']}")
        print(f"\nQUESTION:\n{q['question']}")
        print(f"\nKEY CONCEPTS: {', '.join(q['key_concepts'])}")

        print(f"\nGetting model response...")
        response = ask_model(q["question"])
        print(f"\nMODEL RESPONSE:\n{response[:600]}{'...' if len(response) > 600 else ''}")

        print(f"\nANSWER KEY:\n{q['answer_key']}")
        print(f"\nScoring guide: 2=Fully correct | 1=Partial | 0=Incorrect")

        while True:
            score_input = input(f"Score (0/1/2): ").strip()
            if score_input in ["0", "1", "2"]:
                score = int(score_input)
                break
            print("Enter 0, 1, or 2")

        notes = input("Notes (optional, press Enter to skip): ").strip()

        result = {
            "id": q["id"],
            "category": q["category"],
            "difficulty": q["difficulty"],
            "question": q["question"],
            "model_response": response,
            "answer_key": q["answer_key"],
            "score": score,
            "max_score": 2,
            "notes": notes,
        }
        results.append(result)
        scores_by_category[q["category"]].append(score)

        print(f"Recorded: {score}/2")

    # Calculate scores
    print(f"\n{'='*60}")
    print("CRICKETBENCH v0.1 RESULTS")
    print(f"{'='*60}")

    category_percentages = {}
    for cat, scores in scores_by_category.items():
        if scores:
            pct = sum(scores) / (len(scores) * 2) * 100
            category_percentages[cat] = pct
            print(f"  {cat:<25} {pct:.0f}%  ({sum(scores)}/{len(scores)*2})")

    weighted_total = sum(
        category_percentages.get(cat, 0) * weight
        for cat, weight in CATEGORY_WEIGHTS.items()
    )
    print(f"\n  {'OVERALL (WEIGHTED)':<25} {weighted_total:.0f}%")
    print(f"{'='*60}")
    print(f"\nUpdate these scores in:")
    print(f"  - article/cricket_nemotron_article.md")
    print(f"  - README.md model card section")

    # Save results
    output = {
        "evaluation_date": datetime.now().isoformat(),
        "model": MODEL_NAME,
        "total_questions": len(questions),
        "category_scores": category_percentages,
        "weighted_overall": weighted_total,
        "results": results,
    }

    with open(RESULTS_FILE, "w") as f:
        json.dump(output, f, indent=2)

    print(f"\nFull results saved to {RESULTS_FILE}")


if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
07_upload_to_hf.py
Uploads merged model, GGUF, and dataset to Hugging Face.
Run ON RUNPOD after merge completes.
Run: python scripts/07_upload_to_hf.py
Requires: HF_TOKEN and HF_USERNAME in environment
"""

import os
import sys
from pathlib import Path
from dotenv import load_dotenv
from huggingface_hub import HfApi, create_repo

load_dotenv()

HF_TOKEN = os.getenv("HF_TOKEN")
HF_USERNAME = os.getenv("HF_USERNAME", "brettleehari")

if not HF_TOKEN:
    print("ERROR: HF_TOKEN not set. Get from https://huggingface.co/settings/tokens")
    sys.exit(1)

MODEL_REPO = f"{HF_USERNAME}/cricketmind-nemotron-mini"
DATASET_REPO = f"{HF_USERNAME}/cricketbench-v1"

MERGED_DIR = "./cricketmind-merged"
GGUF_FILE = "./cricketmind-q4.gguf"
TRAIN_FILE = "./data/train.json"
VAL_FILE = "./data/val.json"

api = HfApi(token=HF_TOKEN)


def main():
    print("=" * 60)
    print("CricketMind — Hugging Face Upload")
    print("=" * 60)

    # Create repos
    print(f"Creating model repo: {MODEL_REPO}")
    try:
        create_repo(MODEL_REPO, token=HF_TOKEN, private=False, exist_ok=True)
    except Exception as e:
        print(f"Repo creation note: {e}")

    print(f"Creating dataset repo: {DATASET_REPO}")
    try:
        create_repo(DATASET_REPO, repo_type="dataset", token=HF_TOKEN,
                    private=False, exist_ok=True)
    except Exception as e:
        print(f"Repo creation note: {e}")

    # Upload merged model
    if Path(MERGED_DIR).exists():
        print(f"\nUploading merged model from {MERGED_DIR}...")
        api.upload_folder(
            folder_path=MERGED_DIR,
            repo_id=MODEL_REPO,
            repo_type="model",
        )
        print("  ✓ Merged model uploaded")
    else:
        print(f"  ✗ {MERGED_DIR} not found — skipping merged model upload")

    # Upload GGUF
    if Path(GGUF_FILE).exists():
        gguf_size = Path(GGUF_FILE).stat().st_size / 1e9
        print(f"\nUploading GGUF ({gguf_size:.2f} GB)...")
        api.upload_file(
            path_or_fileobj=GGUF_FILE,
            path_in_repo="cricketmind-q4.gguf",
            repo_id=MODEL_REPO,
            repo_type="model",
        )
        print("  ✓ GGUF uploaded")
    else:
        print(f"  ✗ {GGUF_FILE} not found — skipping GGUF upload")

    # Upload dataset
    for fname, repo_path in [(TRAIN_FILE, "train.json"), (VAL_FILE, "val.json")]:
        if Path(fname).exists():
            print(f"\nUploading {repo_path}...")
            api.upload_file(
                path_or_fileobj=fname,
                path_in_repo=repo_path,
                repo_id=DATASET_REPO,
                repo_type="dataset",
            )
            print(f"  ✓ {repo_path} uploaded")
        else:
            print(f"  ✗ {fname} not found")

    print(f"\n{'='*60}")
    print("UPLOAD COMPLETE")
    print(f"  Model:   https://huggingface.co/{MODEL_REPO}")
    print(f"  Dataset: https://huggingface.co/{DATASET_REPO}")
    print(f"\nNext steps:")
    print(f"  1. Go to HF and write the model card (README.md in the repo)")
    print(f"  2. Update article metrics with real CricketBench scores")
    print(f"  3. Publish article to Substack and Medium")
    print(f"{'='*60}")


if __name__ == "__main__":
    main()
```
